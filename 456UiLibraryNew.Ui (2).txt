local function makeDraggable(topBarObject, object)
	local dragging = nil
	local dragInput = nil
	local dragStart = nil
	local startPosition = nil
	topBarObject.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = input.Position
			startPosition = object.Position
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
				end
			end)
		end
	end)
	topBarObject.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput = input
		end
	end)
	uis.InputChanged:Connect(function(input)
		if input == dragInput and dragging then
			local delta = input.Position - dragStart
			if not djtmemay and cac then
				TweenService:Create(object, TweenInfo.new(DisableAnimation and 0 or 0.35, Enum.EasingStyle.Linear, Enum.EasingDirection.Out), {
					Position = UDim2.new(startPosition.X.Scale, startPosition.X.Offset + delta.X, startPosition.Y.Scale, startPosition.Y.Offset + delta.Y)
				}):Play()
			elseif not djtmemay and not cac then
				object.Position = UDim2.new(startPosition.X.Scale, startPosition.X.Offset + delta.X, startPosition.Y.Scale, startPosition.Y.Offset + delta.Y)
			end
		end
	end)
end

if true then
	local button = btnHide
	local UIS = game:GetService("UserInputService")

	local dragging = false
	local dragInput, dragStart, startPos
	local holdTime = 0.1
	local holdStarted = 0

	local function update(input)
		local delta = input.Position - dragStart
		button.Position = UDim2.new(
			startPos.X.Scale, startPos.X.Offset + delta.X,
			startPos.Y.Scale, startPos.Y.Offset + delta.Y
		)
	end

	local function onInputBegan(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			holdStarted = tick()
			dragStart = input.Position
			startPos = button.Position

			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
					holdStarted = 0
				end
			end)
		end
	end

	local function onInputEnded(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = false
			holdStarted = 0
		end
	end

	local function onInputChanged(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput = input
		end
	end

	button.InputBegan:Connect(onInputBegan)
	button.InputEnded:Connect(onInputEnded)
	button.InputChanged:Connect(onInputChanged)

	RunService.RenderStepped:Connect(function()
		if holdStarted > 0 and (tick() - holdStarted >= holdTime) and not dragging then
			dragging = true
		end

		if dragging and dragInput then
			update(dragInput)
		end
	end)

end

    if not GlobalSearch then
        GlobalSearch = function(searchText)
            searchText = string.lower(searchText)

            if searchText == "" then
                for _, control in pairs(getgenv().AllControls) do
                    if control.TabButton then control.TabButton.Visible = true end
                    if control.Section then control.Section.Visible = true end
                    if control.Element then control.Element.Visible = true end
                end
                for _, tab in pairs(ControlList:GetChildren()) do
                    if not tab:IsA('UIListLayout') then
                        tab.Visible = true
                    end
                end
                return
            end

            for _, control in pairs(getgenv().AllControls) do
                if control.Section then control.Section.Visible = false end
                if control.Element then control.Element.Visible = false end
            end

            for _, tab in pairs(ControlList:GetChildren()) do
                if not tab:IsA('UIListLayout') then
                    tab.Visible = false
                end
            end

            local sectionsWithElements = {}
            local elementsInSection = {}

            for _, control in pairs(getgenv().AllControls) do
                local elementName = string.lower(control.Name or "")
                local sectionName = string.lower(control.SectionName or "")

                local elementFound = string.find(elementName, searchText, 1, true) ~= nil
                local sectionFound = string.find(sectionName, searchText, 1, true) ~= nil

                if not elementsInSection[control.Section] then
                    elementsInSection[control.Section] = {}
                end
                table.insert(elementsInSection[control.Section], {
                    control = control,
                    elementFound = elementFound,
                    sectionFound = sectionFound
                })

                if elementFound then
                    sectionsWithElements[control.Section] = true
                end
            end

            local foundTabs = {}

            for section, elements in pairs(elementsInSection) do
                local shouldShowSection = false
                local hasElementMatch = false

                for _, elementInfo in ipairs(elements) do
                    if elementInfo.sectionFound then
                        shouldShowSection = true
                    end
                    if elementInfo.elementFound then
                        hasElementMatch = true
                    end
                end

                for _, elementInfo in ipairs(elements) do
                    local control = elementInfo.control

                    if elementInfo.elementFound then
                        if control.Element then control.Element.Visible = true end

                        if elementInfo.sectionFound or hasElementMatch then
                            if control.Section then control.Section.Visible = true end
                        end

                        foundTabs[control.TabName] = true
                        if control.TabButton then control.TabButton.Visible = true end
                    elseif elementInfo.sectionFound and not hasElementMatch then
                        if control.Section then control.Section.Visible = true end
                        if control.Element then control.Element.Visible = false end

                        foundTabs[control.TabName] = true
                        if control.TabButton then control.TabButton.Visible = true end
                    end
                end
            end

            for tabName, _ in pairs(foundTabs) do
                for _, tab in pairs(ControlList:GetChildren()) do
                    if not tab:IsA('UIListLayout') and string.find(tab.Name, tabName, 1, true) then
                        tab.Visible = true
                    end
                end
            end

            if not next(foundTabs) then
            end
        end
    end

					local rounded = Precision > 0 and math.floor(value * (10 ^ Precision) + 0.5) / (10 ^ Precision) or math.floor(value + 0.5)
					currentValue = rounded

					TweenService:Create(SliderBarFill, TweenInfo.new(getgenv().UIColor["Tween Animation 1 Speed"]), {

					}):Play()
					TweenService:Create(SliderDrag, TweenInfo.new(getgenv().UIColor["Tween Animation 1 Speed"]), {
						Position = UDim2.new(percent, 0, 0.5, 0)

				end

					local barPos = SliderBar.AbsolutePosition.X
					local barWidth = SliderBar.AbsoluteSize.X
					local mouseX = typeof(input) == "Vector2" and input.X or input.Position.X
					local percent = math.clamp((mouseX - barPos) / barWidth, 0, 1)

						KeybindButton.Text = getKeyName(currentKey)
					end
					TweenService:Create(KeybindHighlight, TweenInfo.new(0.15), {
						BackgroundTransparency = 1
					}):Play()
				end
				
				KeybindButton.MouseButton1Click:Connect(function()
					if listening then
						stopListening()
					else
						startListening()
					end
				end)
				
				uis.InputBegan:Connect(function(input, gameProcessed)
					if listening and not gameProcessed then
						if input.KeyCode ~= Enum.KeyCode.Unknown then
							stopListening(input.KeyCode)
						elseif input.UserInputType == Enum.UserInputType.MouseButton1 then
							stopListening("Mouse1")
						elseif input.UserInputType == Enum.UserInputType.MouseButton2 then
							stopListening("Mouse2")
						elseif input.UserInputType == Enum.UserInputType.MouseButton3 then
							stopListening("Mouse3")
						end
					end
				end)
				
				local keybindFunction = {}
				function keybindFunction.SetKey(key)
					stopListening(key)
				end
				function keybindFunction.GetKey()
					return currentKey
				end
				
				local controlData = {
					Name = Title,
					Section = Section,
					Element = KeybindFrame,
					SectionName = Section_Name,
					TabName = Page_Name,
					TabButton = PageName
				}
				table.insert(getgenv().AllControls, controlData)
				
				return keybindFunction
			end

			-- ==================== COLOR PICKER ====================
			function sectionFunction:AddColorPicker(Setting)
				local Title = Setting.Title or Setting.Text or ""
				local Default = Setting.Default or Color3.fromRGB(255, 255, 255)
				local Callback = Setting.Callback or function() end
				
				local ColorFrame = Instance.new("Frame")
				local ColorBG = Instance.new("Frame")
				local ColorCorner = Instance.new("UICorner")
				local ColorTitle = Instance.new("TextLabel")
				local ColorDisplay = Instance.new("Frame")
				local DisplayCorner = Instance.new("UICorner")
				local ColorButton = Instance.new("TextButton")
				
				local currentColor = Default
				local colorPickerFrame = nil
				local isPickerOpen = false
				local hueBar, satPicker, okBtn
				
				ColorFrame.Name = "ColorFrame"
				ColorFrame.Parent = Section
				ColorFrame.AutomaticSize = Enum.AutomaticSize.Y
				ColorFrame.BackgroundColor3 = Color3.fromRGB(163,162,165)
				ColorFrame.BackgroundTransparency = 1
				ColorFrame.Size = UDim2.new(1, 0, 0, 0)
				
				ColorBG.Name = "ColorBG"
				ColorBG.Parent = ColorFrame
				ColorBG.AnchorPoint = Vector2.new(0.5, 0.5)
				ColorBG.AutomaticSize = Enum.AutomaticSize.Y
				ColorBG.BackgroundColor3 = getgenv().UIColor["Background 1 Color"]
				ColorBG.BackgroundTransparency = getgenv().UIColor["Background 1 Transparency"]
				ColorBG.Position = UDim2.new(0.5, 0, 0.5, 0)
				ColorBG.Size = UDim2.new(1, -10, 0, 0)
				
				ColorCorner.CornerRadius = UDim.new(0, 6)
				ColorCorner.Parent = ColorBG
				
				ColorTitle.Name = "ColorTitle"
				ColorTitle.Parent = ColorBG
				ColorTitle.BackgroundColor3 = Color3.fromRGB(163,162,165)
				ColorTitle.BackgroundTransparency = 1
				ColorTitle.Position = UDim2.new(0, 12, 0, 0)
				ColorTitle.Size = UDim2.new(1, -60, 0, 40)
				ColorTitle.Font = Enum.Font.GothamBold
				ColorTitle.Text = Title
				ColorTitle.TextColor3 = getgenv().UIColor["Text Color"]
				ColorTitle.TextSize = 14
				ColorTitle.TextXAlignment = Enum.TextXAlignment.Left
				ColorTitle.TextYAlignment = Enum.TextYAlignment.Center
				
				ColorDisplay.Name = "ColorDisplay"
				ColorDisplay.Parent = ColorBG
				ColorDisplay.AnchorPoint = Vector2.new(1, 0.5)
				ColorDisplay.BackgroundColor3 = currentColor
				ColorDisplay.Position = UDim2.new(1, -12, 0.5, 0)
				ColorDisplay.Size = UDim2.new(0, 40, 0, 30)
				
				DisplayCorner.CornerRadius = UDim.new(0, 4)
				DisplayCorner.Parent = ColorDisplay
				
				ColorButton.Name = "ColorButton"
				ColorButton.Parent = ColorBG
				ColorButton.BackgroundColor3 = Color3.fromRGB(163,162,165)
				ColorButton.BackgroundTransparency = 1
				ColorButton.Size = UDim2.new(1, 0, 0, 40)
				ColorButton.Font = Enum.Font.SourceSans
				ColorButton.Text = ""
				ColorButton.TextColor3 = Color3.fromRGB(0, 0, 0)
				ColorButton.TextSize = 14
				
				local function updateColorDisplay(color)
					currentColor = color
					ColorDisplay.BackgroundColor3 = color
					pcall(Callback, color)
				end
				
				local function createColorPicker()
					local picker = Instance.new("Frame")
					picker.Name = "ColorPicker"
					picker.Parent = ColorBG
					picker.AnchorPoint = Vector2.new(1, 0)
					picker.BackgroundColor3 = getgenv().UIColor["Background 2 Color"]
					picker.Position = UDim2.new(1, 0, 0, 45)
					picker.Size = UDim2.new(0, 200, 0, 150)
					picker.Visible = false
					picker.ZIndex = 10
					
					local pickerCorner = Instance.new("UICorner", picker)
					pickerCorner.CornerRadius = UDim.new(0, 6)
					
					local pickerStroke = Instance.new("UIStroke", picker)
					pickerStroke.Color = getgenv().UIColor["Border Color"]
					pickerStroke.Thickness = 1
					
					local hue = Instance.new("Frame", picker)
					hue.Name = "HueBar"
					hue.Position = UDim2.new(0, 10, 0, 10)
					hue.Size = UDim2.new(0, 20, 1, -20)
					
					local hueCorner = Instance.new("UICorner", hue)
					hueCorner.CornerRadius = UDim.new(0, 4)
					
					local hueGrad = Instance.new("UIGradient", hue)
					hueGrad.Rotation = 90
					
					local sat = Instance.new("Frame", picker)
					sat.Name = "SatPicker"
					sat.Position = UDim2.new(0, 35, 0, 10)
					sat.Size = UDim2.new(1, -45, 1, -20)
					
					local satCorner = Instance.new("UICorner", sat)
					satCorner.CornerRadius = UDim.new(0, 4)
					
					local ok = Instance.new("TextButton", picker)
					ok.Position = UDim2.new(0, 10, 1, -30)
					ok.Size = UDim2.new(1, -20, 0, 25)
					ok.BackgroundColor3 = getgenv().UIColor["Button Color"]
					ok.Font = Enum.Font.GothamBold
					ok.Text = "OK"
					ok.TextColor3 = Color3.fromRGB(255, 255, 255)
					ok.TextSize = 13
					
					local okCorner = Instance.new("UICorner", ok)
					okCorner.CornerRadius = UDim.new(0, 4)
					
					return picker, hue, sat, ok
				end
				
				ColorButton.MouseButton1Click:Connect(function()
					if isPickerOpen and colorPickerFrame then
						colorPickerFrame.Visible = false
						isPickerOpen = false
					else
						if not colorPickerFrame then
							colorPickerFrame, hueBar, satPicker, okBtn = createColorPicker()
							
							local hueValue = 0
							local satValue = 1
							local valValue = 1
							
							local function updateSatPicker()
								local color = Color3.fromHSV(hueValue, satValue, valValue)
								satPicker.BackgroundColor3 = color
							end
							
							local function updateHueGradient()
								local colors = {}
								for i = 0, 1, 0.05 do
									table.insert(colors, ColorSequenceKeypoint.new(i, Color3.fromHSV(i, 1, 1)))
								end
								hueBar.UIGradient.Color = ColorSequence.new(colors)
							end
							updateHueGradient()
							
							local function getMousePosOnBar(bar, input)
								local barPos = bar.AbsolutePosition
								local barSize = bar.AbsoluteSize
								local mouseY = math.clamp(input.Position.Y - barPos.Y, 0, barSize.Y)
								return mouseY / barSize.Y
							end
							
							local function getMousePosOnSat(sat, input)
								local satPos = sat.AbsolutePosition
								local satSize = sat.AbsoluteSize
								local mouseX = math.clamp(input.Position.X - satPos.X, 0, satSize.X)
								local mouseY = math.clamp(input.Position.Y - satPos.Y, 0, satSize.Y)
								return mouseX / satSize.X, mouseY / satSize.Y
							end
							
							hueBar.InputBegan:Connect(function(input)
								if input.UserInputType == Enum.UserInputType.MouseButton1 then
									local percent = getMousePosOnBar(hueBar, input)
									hueValue = percent
									updateSatPicker()
									updateColorDisplay(Color3.fromHSV(hueValue, satValue, valValue))
								end
							end)
							
							hueBar.InputChanged:Connect(function(input)
								if input.UserInputType == Enum.UserInputType.MouseMovement and uis:IsMouseButtonPressed(Enum.UserInputType.MouseButton1) then
									local percent = getMousePosOnBar(hueBar, input)
									hueValue = percent
									updateSatPicker()
									updateColorDisplay(Color3.fromHSV(hueValue, satValue, valValue))
								end
							end)
							
							satPicker.InputBegan:Connect(function(input)
								if input.UserInputType == Enum.UserInputType.MouseButton1 then
									local x, y = getMousePosOnSat(satPicker, input)
									satValue = x
									valValue = 1 - y
									updateColorDisplay(Color3.fromHSV(hueValue, satValue, valValue))
								end
							end)
							
							satPicker.InputChanged:Connect(function(input)
								if input.UserInputType == Enum.UserInputType.MouseMovement and uis:IsMouseButtonPressed(Enum.UserInputType.MouseButton1) then
									local x, y = getMousePosOnSat(satPicker, input)
									satValue = x
									valValue = 1 - y
									updateColorDisplay(Color3.fromHSV(hueValue, satValue, valValue))
								end
							end)
							
							okBtn.MouseButton1Click:Connect(function()
								colorPickerFrame.Visible = false
								isPickerOpen = false
							end)
						end
						
						colorPickerFrame.Visible = true
						isPickerOpen = true
					end
				end)
				
				local colorFunction = {}
				function colorFunction.SetColor(color)
					updateColorDisplay(color)
				end
				function colorFunction.GetColor()
					return currentColor
				end
				
				local controlData = {
					Name = Title,
					Section = Section,
					Element = ColorFrame,
					SectionName = Section_Name,
					TabName = Page_Name,
					TabButton = PageName
				}
				table.insert(getgenv().AllControls, controlData)
				
				return colorFunction
			end

			return sectionFunction
		end

		return pageFunction
